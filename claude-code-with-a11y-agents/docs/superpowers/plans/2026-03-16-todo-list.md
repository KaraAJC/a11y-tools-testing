# Todo List Website Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single self-contained `todo.html` file with add, inline-edit, delete, priority management, and sort-by-priority — fully WCAG AA accessible.

**Architecture:** State lives in a `todos` array with a `render()` function that rebuilds the list on every change. No framework, no build step. All HTML, CSS, and JS in one file. Feature-by-feature implementation with manual browser verification after each task.

**Tech Stack:** Vanilla HTML5 + CSS3 + JavaScript (ES6). No dependencies. Open directly in any browser.

**Spec:** `docs/superpowers/specs/2026-03-16-todo-list-design.md`

---

## File Structure

| File | Responsibility |
|---|---|
| `todo.html` | Everything — markup, styles, and logic in one self-contained file |

All sections live in `todo.html`:
- `<head>`: meta, title, `<style>` block (all CSS)
- `<body>`: semantic HTML structure
- `<script>` (bottom of body): state variables, `render()`, event handlers

```html
  <main>
    <h1>My Todos</h1>

  <script>
    // ─── State ───────────────────────────────────────────────────────────────
    let todos = [];
    let sortEnabled = false;
    let editingId = null;
    let nextId = 1;

    // ─── Helpers ─────────────────────────────────────────────────────────────
    const PRIORITY_ORDER = { now: 0, 'this-week': 1, later: 2 };
    const PRIORITY_LABELS = { now: 'NOW', 'this-week': 'THIS WEEK', later: 'LATER' };
    const PRIORITY_HUMAN  = { now: 'Now', 'this-week': 'This Week', later: 'Later' };

    function announce(msg) {
      const region = document.getElementById('live-region');
      region.textContent = '';
      // Force re-announcement even for identical messages
      requestAnimationFrame(() => { region.textContent = msg; });
    }

    function getSortedTodos() {
      if (!sortEnabled) return [...todos];
      return [...todos].sort((a, b) => PRIORITY_ORDER[a.priority] - PRIORITY_ORDER[b.priority]);
    }

    // ─── Render ──────────────────────────────────────────────────────────────
    function render() {
      const list = document.getElementById('todo-list');
      list.innerHTML = '';

      const displayed = getSortedTodos();

      if (displayed.length === 0) {
        const empty = document.createElement('li');
        empty.className = 'empty-state';
        empty.textContent = 'No todos yet. Add one above.';
        list.appendChild(empty);
        return;
      }

      displayed.forEach(todo => {
        const li = buildTodoItem(todo);
        list.appendChild(li);
      });
    }

    function buildTodoItem(todo) {
      const li = document.createElement('li');
      li.dataset.id = todo.id;

      if (editingId === todo.id) {
        buildEditRow(li, todo);
      } else {
        buildViewRow(li, todo);
      }

      return li;
    }

    function buildViewRow(li, todo) {
      // Text
      const span = document.createElement('span');
      span.className = 'todo-text';
      span.textContent = todo.text;

      // Priority badge button
      const priorityBtn = document.createElement('button');
      priorityBtn.className = 'priority-btn';
      priorityBtn.dataset.priority = todo.priority;
      priorityBtn.setAttribute('aria-label', `Change priority for ${todo.text}, currently ${PRIORITY_HUMAN[todo.priority]}`);
      priorityBtn.innerHTML = `<span aria-hidden="true">${PRIORITY_LABELS[todo.priority]}</span>`;
      priorityBtn.addEventListener('click', () => startPriorityChange(todo.id));

      // Edit button
      const editBtn = document.createElement('button');
      editBtn.className = 'icon-btn';
      editBtn.setAttribute('aria-label', `Edit ${todo.text}`);
      editBtn.innerHTML = '<span aria-hidden="true">✏</span>';
      editBtn.addEventListener('click', () => startEdit(todo.id));

      // Delete button
      const deleteBtn = document.createElement('button');
      deleteBtn.className = 'icon-btn delete-btn';
      deleteBtn.setAttribute('aria-label', `Delete ${todo.text}`);
      deleteBtn.innerHTML = '<span aria-hidden="true">✕</span>';
      deleteBtn.addEventListener('click', () => deleteTodo(todo.id));

      li.append(span, priorityBtn, editBtn, deleteBtn);
    }

    function buildEditRow(li, todo) {
      const errorId = `edit-error-${todo.id}`;

      // Edit input
      const input = document.createElement('input');
      input.type = 'text';
      input.className = 'edit-input';
      input.value = todo.text;
      input.setAttribute('aria-label', `Editing: ${todo.text}`);
      input.setAttribute('aria-describedby', errorId);
      input.setAttribute('autocomplete', 'off');

      // Error message (hidden until needed)
      const error = document.createElement('span');
      error.id = errorId;
      error.className = 'edit-error sr-only';
      error.setAttribute('aria-live', 'assertive');
      error.textContent = '';

      // Save button
      const saveBtn = document.createElement('button');
      saveBtn.className = 'save-btn';
      saveBtn.textContent = 'Save';
      saveBtn.setAttribute('aria-label', `Save edit for ${todo.text}`);

      // Cancel button
      const cancelBtn = document.createElement('button');
      cancelBtn.className = 'cancel-btn';
      cancelBtn.textContent = 'Cancel';
      cancelBtn.setAttribute('aria-label', `Cancel edit for ${todo.text}`);

      function attemptSave() {
        const trimmed = input.value.trim();
        if (!trimmed) {
          input.setAttribute('aria-invalid', 'true');
          error.textContent = 'Todo text cannot be empty.';
          error.classList.remove('sr-only');
          input.focus();
          return;
        }
        const todo_ref = todos.find(t => t.id === todo.id);
        if (todo_ref) todo_ref.text = trimmed;
        editingId = null;
        render();
        announce(`Updated: ${trimmed}`);
        // Return focus to edit button
        const editBtn = document.querySelector(`[data-id="${todo.id}"] .icon-btn`);
        if (editBtn) editBtn.focus();
      }

      function cancelEdit() {
        editingId = null;
        render();
        const editBtn = document.querySelector(`[data-id="${todo.id}"] .icon-btn`);
        if (editBtn) editBtn.focus();
      }

      input.addEventListener('keydown', (e) => {
        if (e.key === 'Enter') { e.preventDefault(); attemptSave(); }
        if (e.key === 'Escape') { cancelEdit(); }
      });

      saveBtn.addEventListener('click', attemptSave);
      cancelBtn.addEventListener('click', cancelEdit);

      li.append(input, saveBtn, cancelBtn, error);
    }

    // ─── Actions ─────────────────────────────────────────────────────────────
    function addTodo(text, priority) {
      todos.push({ id: nextId++, text, priority });
      render();
      announce(`Added: ${text}`);
    }

    function deleteTodo(id) {
      const list = document.getElementById('todo-list');
      const items = Array.from(list.querySelectorAll('li[data-id]'));
      const idx = items.findIndex(li => Number(li.dataset.id) === id);

      const todo = todos.find(t => t.id === id);
      const deletedText = todo ? todo.text : '';

      todos = todos.filter(t => t.id !== id);
      render();
      announce(`Deleted: ${deletedText}`);

      // Move focus: next item → previous item → add input
      const updatedItems = Array.from(document.querySelectorAll('#todo-list li[data-id]'));
      const target = updatedItems[idx] || updatedItems[idx - 1];
      if (target) {
        const deleteBtn = target.querySelector('.delete-btn');
        if (deleteBtn) deleteBtn.focus();
      } else {
        document.getElementById('new-todo-input').focus();
      }
    }

    function startEdit(id) {
      editingId = id;
      render();
      const input = document.querySelector(`[data-id="${id}"] .edit-input`);
      if (input) input.focus();
    }

    function startPriorityChange(id) {
      const todo = todos.find(t => t.id === id);
      if (!todo) return;

      const li = document.querySelector(`[data-id="${id}"]`);
      if (!li) return;

      const priorityBtn = li.querySelector('.priority-btn');
      if (!priorityBtn) return;

      // Replace badge with inline select
      const sel = document.createElement('select');
      sel.className = 'priority-select';
      sel.setAttribute('aria-label', `Priority for ${todo.text}`);

      ['now', 'this-week', 'later'].forEach(val => {
        const opt = document.createElement('option');
        opt.value = val;
        opt.textContent = PRIORITY_HUMAN[val];
        if (val === todo.priority) opt.selected = true;
        sel.appendChild(opt);
      });

      const originalPriority = todo.priority;

      function commitChange() {
        const newPriority = sel.value;
        todo.priority = newPriority;
        render();
        announce(`Priority changed to ${PRIORITY_HUMAN[newPriority]}`);
        // Return focus to priority badge
        const badge = document.querySelector(`[data-id="${id}"] .priority-btn`);
        if (badge) badge.focus();
      }

      function cancelChange() {
        // Restore original badge without saving
        render();
        const badge = document.querySelector(`[data-id="${id}"] .priority-btn`);
        if (badge) badge.focus();
      }

      sel.addEventListener('change', commitChange);

      sel.addEventListener('keydown', (e) => {
        if (e.key === 'Escape') { e.preventDefault(); cancelChange(); }
      });

      sel.addEventListener('blur', (e) => {
        // If focus moves outside the select without a change, cancel
        if (sel.value === originalPriority) {
          cancelChange();
        }
        // If value changed, commitChange already fired via 'change' event
      });

      priorityBtn.replaceWith(sel);
      sel.focus();
    }

    // ─── Event wiring ────────────────────────────────────────────────────────
    document.getElementById('add-form').addEventListener('submit', (e) => {
      e.preventDefault();
      const input = document.getElementById('new-todo-input');
      const priority = document.getElementById('new-todo-priority').value;
      const text = input.value.trim();
      if (!text) return; // Belt-and-suspenders; browser required handles this
      addTodo(text, priority);
      input.value = '';
      input.focus();
    });

    document.getElementById('sort-btn').addEventListener('click', () => {
      sortEnabled = !sortEnabled;
      const btn = document.getElementById('sort-btn');
      btn.setAttribute('aria-pressed', String(sortEnabled));
      render();
      announce(sortEnabled ? 'Sorted by priority' : 'Original order restored');
    });

    // ─── Init ────────────────────────────────────────────────────────────────
    render();
  </script>
</body>
</html>
```
