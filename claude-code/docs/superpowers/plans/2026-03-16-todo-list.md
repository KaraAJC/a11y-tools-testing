# Todo List Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single self-contained `index.html` todo list app with add, inline-edit, delete, priority badges, and a sort toggle.

**Architecture:** All HTML, CSS, and JavaScript live in one file. A `todos` array and a `sortByPriority` boolean are the only state. Every mutation calls `render()`, which rebuilds the list from scratch. No frameworks, no dependencies, no persistence.

**Tech Stack:** Vanilla HTML5, CSS3, JavaScript (ES6+). Single file. No build tools.

## Chunk 2: State + Add + Render

### Task 3: Add state model and render() function

**Files:**
- Modify: `claude-code/index.html` — `<script>` block

- [ ] **Step 1: Replace the `/* JS goes here */` comment with state variables and render()**

```js
// --- State ---
let todos = [];
let nextId = 1;
let sortByPriority = false;

const PRIORITY_ORDER = { now: 0, 'this-week': 1, later: 2 };
const PRIORITY_LABEL = { now: 'Now', 'this-week': 'This Week', later: 'Later' };
const PRIORITY_BADGE_CLASS = { now: 'badge-now', 'this-week': 'badge-this-week', later: 'badge-later' };
const PRIORITY_ITEM_CLASS = { now: 'priority-now', 'this-week': 'priority-this-week', later: 'priority-later' };
const PRIORITY_CYCLE = { now: 'this-week', 'this-week': 'later', later: 'now' };

// --- Render ---
function render() {
  const list = document.getElementById('todo-list');
  const sortBtn = document.getElementById('sort-toggle');

  sortBtn.textContent = sortByPriority ? 'Remove Sort' : 'Sort by Priority';
  sortBtn.classList.toggle('active', sortByPriority);

  if (todos.length === 0) {
    list.innerHTML = '<li class="empty-message">No todos yet — add one above.</li>';
    return;
  }

  const displayed = sortByPriority
    ? [...todos].sort((a, b) => PRIORITY_ORDER[a.priority] - PRIORITY_ORDER[b.priority])
    : todos;

  list.innerHTML = displayed.map(todo => `
    <li class="todo-item ${PRIORITY_ITEM_CLASS[todo.priority]}" data-id="${todo.id}">
      <span class="todo-text" data-action="edit" data-id="${todo.id}">${escapeHtml(todo.text)}</span>
      <button class="priority-badge ${PRIORITY_BADGE_CLASS[todo.priority]}" data-action="cycle-priority" data-id="${todo.id}">
        ${PRIORITY_LABEL[todo.priority]}
      </button>
      <button class="delete-btn" data-action="delete" data-id="${todo.id}" aria-label="Delete todo">&#x2715;</button>
    </li>
  `).join('');
}

function escapeHtml(str) {
  return str.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

// Initial render
render();
```

document.getElementById('add-form').addEventListener('submit', function(e) {
  e.preventDefault();
  const input = document.getElementById('new-todo-input');
  const text = input.value.trim();
  if (!text) return;
  const priority = document.getElementById('priority-select').value;
  todos.push({ id: nextId++, text, priority, createdAt: Date.now() });
  input.value = '';
  render();
});

// --- Event delegation for todo list actions ---
document.getElementById('todo-list').addEventListener('click', function(e) {
  const action = e.target.dataset.action;
  const id = Number(e.target.dataset.id);
  if (!action || !id) return;

  if (action === 'delete') {
    todos = todos.filter(t => t.id !== id);
    render();
  }

  if (action === 'cycle-priority') {
    const todo = todos.find(t => t.id === id);
    if (todo) {
      todo.priority = PRIORITY_CYCLE[todo.priority];
      render();
    }
  }

  if (action === 'edit') {
    startEdit(id);
  }
});
```

// --- Inline edit ---
function startEdit(id) {
  const todo = todos.find(t => t.id === id);
  if (!todo) return;

  const li = document.querySelector(`.todo-item[data-id="${id}"]`);
  const span = li.querySelector('.todo-text');

  const input = document.createElement('input');
  input.type = 'text';
  input.className = 'todo-edit-input';
  input.value = todo.text;

  li.replaceChild(input, span);
  input.focus();
  input.select();

  function commitEdit() {
    const newText = input.value.trim();
    if (newText) todo.text = newText;
    render();
  }

  input.addEventListener('keydown', function(e) {
    if (e.key === 'Enter') { e.preventDefault(); commitEdit(); }
    if (e.key === 'Escape') { render(); } // cancel — discard changes
  });

  input.addEventListener('blur', commitEdit);
}
```
